pipeline {
    agent any

    environment {
        AZURE_CREDENTIALS = credentials('azure-credentials-id') // Add Azure credentials in Jenkins
        ACR_NAME = 'ML-Model'
        ACR_LOGIN_SERVER = 'acr-login-url'
    }

    stages {
        stage('Checkout Code') {
            steps {
                git branch: 'main', url: 'https://github.com/dhamodharankj/Dhamodharan_Verti_Sytem.git'
            }
        }
        stage('Build Docker Image') {
            steps {
                script {
                    dockerImage = docker.build("${dev.ACR_LOGIN_SERVER}/${dev.CLEAN_DATA_ML_MODEL}:${dev.1.0}")
                }
            }
        }

        stage('Login to Azure ACR and push the image') {
            steps {
                script {
                    sh """
                    az login --service-principal -u ${dev.AZURE_CREDENTIALS_USR} -p ${env.AZURE_CREDENTIALS_PSW}
                    az acr login --name ${env.ACR_NAME}
                    """
					dockerImage.push()
                }
            }
        }

    }    

    post {
        always {
            cleanWs()
        }
    }
}
